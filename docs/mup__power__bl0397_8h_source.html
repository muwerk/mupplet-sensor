65unsigned long getBlpResetpIrqCount(uint8_t irqno) {
66 unsigned long count = (unsigned long)-1;
67 noInterrupts();
68 if (irqno < USTD_MAX_BLP_PIRQS) {
69 count = blp_pirpcounter[irqno];
70 blp_pirpcounter[irqno] = 0;
71 }
72 interrupts();
73 return count;
74}

76double getBlpResetpIrqFrequency(uint8_t irqno, unsigned long minDtUs = 50) {
77 double frequency = 0.0;
78 noInterrupts();
79 if (irqno < USTD_MAX_BLP_PIRQS) {
80 unsigned long count = blp_pirpcounter[irqno];
81 unsigned long dt = timeDiff(blp_pbeginIrqTimer[irqno], blp_plastIrqTimer[irqno]);
82 if (dt > minDtUs) { // Ignore small Irq flukes
83 frequency = (count * 500000.0) / dt; // = count/2.0*1000.0000 uS / dt; no.
84 // of waves (count/2) / dt.
85 }
86 blp_pbeginIrqTimer[irqno] = 0;
87 blp_pirpcounter[irqno] = 0;
88 blp_plastIrqTimer[irqno] = 0;
89 }
90 interrupts();
91 return frequency;
92}

275 void loop() {
276 double watts = getBlpResetpIrqFrequency(interruptIndex_CF) / powerRenormalization *
277 userCalibrationPowerFactor;
278 if ((frequencyCF.lastVal == 0.0 && watts > 0.0) ||
279 (frequencyCF.lastVal > 0.0 && watts == 0.0))
280 frequencyCF.reset();
281 if (watts >= 0.0 && watts < 3800) {
282 if (frequencyCF.filter(&watts)) {
283 CFfrequencyVal = watts;
284 publish_CF();
285 }
286 }
287 double mfreq = getBlpResetpIrqFrequency(interruptIndex_CF1);
288 if (bSELi) {
289 double volts = mfreq / voltageRenormalisation * userCalibrationVoltageFactor;
290 if (volts < 5.0 || (volts >= 100.0 && volts < 260)) {
291 if ((frequencyCF1_V.lastVal == 0.0 && volts > 0.0) ||
292 (frequencyCF1_V.lastVal > 0.0 && volts == 0.0))
293 frequencyCF1_V.reset();
294 if (frequencyCF1_V.filter(&volts)) {
295 CF1_VfrequencyVal = volts;
296 publish_CF1_V();
297 }
298 }
299 } else {
300 double currents = mfreq / currentRenormalisation * userCalibrationCurrentFactor;
301 if (currents >= 0.0 && currents < 16.0) {
302 if ((frequencyCF1_I.lastVal == 0.0 && currents > 0.0) ||
303 (frequencyCF1_I.lastVal > 0.0 && currents == 0.0))
304 frequencyCF1_I.reset();
305 if (frequencyCF1_I.filter(&currents)) {
306 CF1_IfrequencyVal = currents;
307 publish_CF1_I();
308 }
309 }
310 }
311 bSELi = changeBlpSELi(!bSELi, pin_SELi, interruptIndex_CF1);
312 }
