280 void begin(Scheduler *_pSched, BMESampleMode _sampleMode=BMESampleMode::STANDARD, TwoWire *_pWire=&Wire) {
281 pSched = _pSched;
282 pWire=_pWire;
283 uint8_t data;
284
285 setSampleMode(_sampleMode);
286
287 auto ft = [=]() { this->loop(); };
288 tID = pSched->add(ft, name, 500); // 500us
289
290 auto fnall = [=](String topic, String msg, String originator) {
291 this->subsMsg(topic, msg, originator);
292 };
293 pSched->subscribe(tID, name + "/sensor/#", fnall);
294
295 lastError=i2c_checkAddress(i2c_address);
296 if (lastError==BMEError::OK) {
297 if (!i2c_readRegisterByte(0xd0, &data)) { // 0xd0: chip-id register
298#ifdef USE_SERIAL_DBG
299 Serial.print("Failed to inquire BME280 chip-id at address 0x");
300 Serial.println(i2c_address, HEX);
301#endif
302 bActive=false;
303 } else {
304 if (data==0x60) { // 0x60: signature of BME280
305 if (!initBmeSensorConstants()) {
306#ifdef USE_SERIAL_DBG
307 Serial.print("Failed to read calibration data for sensor BME280 at address 0x");
308 Serial.println(i2c_address, HEX);
309#endif
310 lastError=BMEError::I2C_CALIBRATION_READ_FAILURE;
311 bActive = false;
312 } else {
313#ifdef USE_SERIAL_DBG
314 Serial.print("BME280 sensor active at address 0x");
315 Serial.println(i2c_address, HEX);
316#endif
317 sensorState=BMESensorState::IDLE;
318 bActive=true;
319 }
320 } else {
321#ifdef USE_SERIAL_DBG
322 Serial.print("Wrong hardware (not BME280) found at address 0x");
323 Serial.print(i2c_address, HEX);
324 Serial.print(" chip-id is ");
325 Serial.print(data, HEX);
326 Serial.println(" expected: 0x60 for BME280.");
327 if (data==0x58) Serial.println("This is not a BME280 but a BME280 sensor (no humidity). This might be a fake chip/source.");
328#endif
329 lastError=BMEError::I2C_WRONG_HARDWARE_AT_ADDRESS;
330 bActive = false;
331 }
332 }
333 } else {
334#ifdef USE_SERIAL_DBG
335 Serial.print("No BME280 sensor found at address 0x");
336 Serial.println(i2c_address, HEX);
337#endif
338 bActive = false;
339 }
340 }

385 BMEError i2c_checkAddress(uint8_t address) {
386 pWire->beginTransmission(address);
387 byte error = pWire->endTransmission();
388 if (error == 0) {
389 lastError=BMEError::OK;
390 return lastError;
391 } else if (error == 4) {
392 lastError=I2C_HW_ERROR;
393 }
394 lastError=I2C_DEVICE_NOT_AT_ADDRESS;
395 return lastError;

398 bool i2c_endTransmission(bool releaseBus) {
399 uint8_t retCode=pWire->endTransmission(releaseBus); // true: release bus, send stop
400 switch (retCode) {
401 case 0:
402 lastError=BMEError::OK;
403 return true;
404 case 1:
405 lastError=BMEError::I2C_WRITE_DATA_TOO_LONG;
406 return false;
407 case 2:
408 lastError=BMEError::I2C_WRITE_NACK_ON_ADDRESS;
409 return false;
410 case 3:
411 lastError=BMEError::I2C_WRITE_NACK_ON_DATA;
412 return false;
413 case 4:
414 lastError=BMEError::I2C_WRITE_ERR_OTHER;
415 return false;
416 case 5:
417 lastError=BMEError::I2C_WRITE_TIMEOUT;
418 return false;
419 default:
420 lastError=BMEError::I2C_WRITE_INVALID_CODE;
421 return false;
422 }
423 return false;
424 }

426 bool i2c_readRegisterByte(uint8_t reg, uint8_t* pData) {
427 *pData=(uint8_t)-1;
428 pWire->beginTransmission(i2c_address);
429 if (pWire->write(&reg,1)!=1) {
430 lastError=BMEError::I2C_REGISTER_WRITE_ERROR;
431 return false;
432 }
433 if (i2c_endTransmission(true)==false) return false;
434 uint8_t read_cnt=pWire->requestFrom(i2c_address,(uint8_t)1,(uint8_t)true);
435 if (read_cnt!=1) {
436 lastError=I2C_READ_REQUEST_FAILED;
437 return false;
438 }
439 *pData=pWire->read();
440 return true;
441 }

443 bool i2c_readRegisterWord(uint8_t reg, uint16_t* pData) {
444 *pData=(uint16_t)-1;
445 pWire->beginTransmission(i2c_address);
446 if (pWire->write(&reg,1)!=1) {
447 lastError=BMEError::I2C_REGISTER_WRITE_ERROR;
448 return false;
449 }
450 if (i2c_endTransmission(true)==false) return false;
451 uint8_t read_cnt=pWire->requestFrom(i2c_address,(uint8_t)2,(uint8_t)true);
452 if (read_cnt!=2) {
453 lastError=I2C_READ_REQUEST_FAILED;
454 return false;
455 }
456 uint8_t hb=pWire->read();
457 uint8_t lb=pWire->read();

609 bool sensorStateMachine() {
610 bool newData=false;
611 uint32_t rt;
612 uint32_t rp;
613 uint16_t rh;
614 uint8_t reg_data;
615 const uint8_t status_register=0xf3;
616 const uint8_t measure_mode_register=0xf4;
617 const uint8_t config_register=0xf5;
618 const uint8_t ctrl_hum_register=0xf2;
619 const uint8_t temperature_registers=0xfa;
620 const uint8_t pressure_registers=0xf7;
621 const uint8_t humidity_registers=0xfd;
622 uint8_t status;
623 uint8_t normalmodeInactivity=0, IIRfilter=0; // not used.
624 switch (sensorState) {
625 case BMESensorState::UNAVAILABLE:
626 break;
627 case BMESensorState::IDLE:
628 reg_data=(normalmodeInactivity<<5) + (IIRfilter << 2) + 0;
629 if (!i2c_writeRegisterByte(config_register, reg_data)) {
630 ++errs;
631 sensorState=BMESensorState::WAIT_NEXT_MEASUREMENT;
632 stateMachineClock=micros();
633 break;
634 }
635 reg_data=oversampleModeHumidity & 0x7;
636 if (!i2c_writeRegisterByte(ctrl_hum_register, reg_data)) {
637 ++errs;
638 sensorState=BMESensorState::WAIT_NEXT_MEASUREMENT;
639 stateMachineClock=micros();
640 break;
641 }
642 reg_data=(oversampleModeTemperature << 5) + (oversampleModePressure << 2) + 0x1; // 0x3: normal mode, 0x1 one-shot
643 if (!i2c_writeRegisterByte(measure_mode_register, reg_data)) {
644 ++errs;
645 sensorState=BMESensorState::WAIT_NEXT_MEASUREMENT;
646 stateMachineClock=micros();
647 break;
648 } else { // Start temperature read
649 sensorState=BMESensorState::MEASUREMENT_WAIT;
650 stateMachineClock=micros();
651 }
652 break;
653 case BMESensorState::MEASUREMENT_WAIT:
654 if (!i2c_readRegisterByte(status_register, &status)) {
655 // no status
656 ++errs;
657 sensorState=BMESensorState::WAIT_NEXT_MEASUREMENT;
658 stateMachineClock=micros();
659 break;
660 }
661 status = status & 0x09;
662 if (timeDiff(stateMachineClock,micros()) > 1 && status==0) { // 1ms for meas, no status set.
663 rt=0; rp=0;
664 if (i2c_readRegisterTripple(temperature_registers,&rt) && i2c_readRegisterTripple(pressure_registers, &rp) &&
665 i2c_readRegisterWord(humidity_registers, &rh)) {
666 rawTemperature=rt>>4;
667 rawPressure=rp>>4;
668 rawHumidity=(int32_t)rh;
669 sensorState=BMESensorState::WAIT_NEXT_MEASUREMENT;
670 stateMachineClock=micros();
671 ++oks;
672 newData=true;
673 } else {
674 ++errs;
675 sensorState=BMESensorState::WAIT_NEXT_MEASUREMENT;
676 stateMachineClock=micros();
677 }
678 }
679 break;
680 case BMESensorState::WAIT_NEXT_MEASUREMENT:
681 if (timeDiff(stateMachineClock,micros()) > pollRateUs) {
682 sensorState=BMESensorState::IDLE; // Start next cycle.
683 }
684 break;
685 }
686 return newData;
687 }

689 // From BOSCH datasheet BMA150 data sheet Rev. 1.4 | taken from BMP280, seems identical to BME280
690 // Returns temperature in DegC, double precision. Output value of 51.23 equals 51.23 DegC.
691 // t_fine carries fine temperature used by pressure
692 //int32_t t_fine;
693 double bme280_compensate_T_double(int32_t adc_T, int32_t *pt_fine) {
694 double var1, var2, T;
695 var1 = (((double)adc_T)/16384.0 - ((double)dig_T1)/1024.0) * ((double)dig_T2);
696 var2 = ((((double)adc_T)/131072.0 - ((double)dig_T1)/8192.0) *
697 (((double)adc_T)/131072.0 - ((double)dig_T1)/8192.0)) * ((double)dig_T3);
698 *pt_fine = (int32_t)(var1 + var2);
699 T = (var1 + var2) / 5120.0;
700 return T;
701 }
702
703 // From BOSCH datasheet BMA150 data sheet Rev. 1.4 | taken from BMP280, seems identical to BME280
704 // Returns pressure in Pa as double. Output value of 96386.2 equals 96386.2 Pa = 963.862 hPa
705 double bme280_compensate_P_double(int32_t adc_P, int32_t t_fine) {
706 double var1, var2, p;
707 var1 = ((double)t_fine/2.0) - 64000.0;
708 var2 = var1 * var1 * ((double)dig_P6) / 32768.0;
709 var2 = var2 + var1 * ((double)dig_P5) * 2.0;
710 var2 = (var2/4.0)+(((double)dig_P4) * 65536.0);
711 var1 = (((double)dig_P3) * var1 * var1 / 524288.0 + ((double)dig_P2) * var1) / 524288.0;
712 var1 = (1.0 + var1 / 32768.0)*((double)dig_P1);
713 if (var1 == 0.0) {
714 return 0; // avoid exception caused by division by zero
715 }
716 p = 1048576.0 - (double)adc_P;
717 p = (p - (var2 / 4096.0)) * 6250.0 / var1;
718 var1 = ((double)dig_P9) * p * p / 2147483648.0;

727 double var_H;
728 var_H = (((double)t_fine) - 76800.0);

730 (((double)dig_H2) / 65536.0 * (1.0 + ((double)dig_H6) / 67108864.0 * var_H *
731 (1.0 + ((double)dig_H3) / 67108864.0 * var_H)));
732 var_H = var_H * (1.0 - ((double)dig_H1) * var_H / 524288.0);
733 if (var_H > 100.0) {
734 var_H = 100.0;
735 } else {
736 if (var_H < 0.0) var_H = 0.0;
737 }
738 return var_H;
