182 void begin(Scheduler *_pSched, TwoWire *_pWire=&Wire) {
183 pSched = _pSched;
184 pWire=_pWire;
185
186 pWire->begin(); // required!
187 auto ft = [=]() { this->loop(); };
188 tID = pSched->add(ft, name, 1000000); // 1s
189
190 auto fnall = [=](String topic, String msg, String originator) {
191 this->subsMsg(topic, msg, originator);
192 };
193 pSched->subscribe(tID, name + "/sensor/#", fnall);
194
195 lastError=i2c_checkAddress(i2c_address);
196 if (lastError==TSLError::OK) {
197 uint8_t id, rev;
198 if (TSLSensorGetRevID(&id, &rev)) {
199 if (id==5 || id==1) { // TSL2561 id should be either 5 (reality) or 1 (datasheet)
200 if (TSLSensorPower(true)) {
201 sensorState=TSLSensorState::IDLE;
202 bActive=true;
203#ifdef USE_SERIAL_DBG
204 Serial.println("TSL2561: Powered on, revision: " + String(rev));
205#endif
206 } else {
207#ifdef USE_SERIAL_DBG
208 Serial.println("TSL2561: Power on failed");
209#endif
210 lastError=TSLError::I2C_WRITE_ERR_OTHER;
211 }
212 } else {
213#ifdef USE_SERIAL_DBG
214 Serial.println("TSL2561: Bad sensor ID: "+String(id)+", expected: 1, revision: "+String(rev));
215#endif
216 lastError=TSLError::I2C_WRONG_HARDWARE_AT_ADDRESS;
217 }
218 } else {
219#ifdef USE_SERIAL_DBG
220 Serial.println("TSL2561: Failed to get sensor ID, wrong hardware?");
221#endif
222 lastError=TSLError::I2C_WRONG_HARDWARE_AT_ADDRESS;
223 }
224 } else {
225#ifdef USE_SERIAL_DBG
226 Serial.println("TSL2561: Failed to check I2C address, wrong address?");
227#endif
228 lastError=TSLError::I2C_DEVICE_NOT_AT_ADDRESS;
229 }

299 TSLError i2c_checkAddress(uint8_t address) {
300 pWire->beginTransmission(address);
301 byte error = pWire->endTransmission();
302 if (error == 0) {
303 lastError=TSLError::OK;
304 return lastError;

312 bool i2c_endTransmission(bool releaseBus) {
313 uint8_t retCode=pWire->endTransmission(releaseBus); // true: release bus, send stop
314 switch (retCode) {
315 case 0:
316 lastError=TSLError::OK;
317 return true;
318 case 1:
319 lastError=TSLError::I2C_WRITE_DATA_TOO_LONG;
320 return false;
321 case 2:
322 lastError=TSLError::I2C_WRITE_NACK_ON_ADDRESS;
323 return false;
324 case 3:
325 lastError=TSLError::I2C_WRITE_NACK_ON_DATA;
326 return false;
327 case 4:
328 lastError=TSLError::I2C_WRITE_ERR_OTHER;
329 return false;
330 case 5:
331 lastError=TSLError::I2C_WRITE_TIMEOUT;
332 return false;
333 default:
334 lastError=TSLError::I2C_WRITE_INVALID_CODE;
335 return false;
336 }
337 return false;

340 bool i2c_readRegisterByte(uint8_t reg, uint8_t* pData) {
341 *pData=(uint8_t)-1;
342 pWire->beginTransmission(i2c_address);
343 if (pWire->write(&reg,1)!=1) {
344 lastError=TSLError::I2C_REGISTER_WRITE_ERROR;
345 return false;
346 }
347 if (i2c_endTransmission(true)==false) return false;
348 uint8_t read_cnt=pWire->requestFrom(i2c_address,(uint8_t)1,(uint8_t)true);
349 if (read_cnt!=1) {
350 lastError=I2C_READ_REQUEST_FAILED;
351 return false;
352 }
353 *pData=pWire->read();
354 return true;
355 }

357 bool i2c_readRegisterWord(uint8_t reg, uint16_t* pData, bool stop=true, bool allow_irqs=true) {
358 *pData=(uint16_t)-1;
359 if (!allow_irqs) noInterrupts();
360 pWire->beginTransmission(i2c_address);
361 if (pWire->write(&reg,1)!=1) {
362 if (!allow_irqs) interrupts();
363 lastError=TSLError::I2C_REGISTER_WRITE_ERROR;
364 return false;
365 }
366 if (i2c_endTransmission(stop)==false) return false;
367 uint8_t read_cnt=pWire->requestFrom(i2c_address,(uint8_t)2,(uint8_t)true);
368 if (read_cnt!=2) {
369 if (!allow_irqs) interrupts();
370 lastError=I2C_READ_REQUEST_FAILED;
371 return false;
372 }
373 if (!allow_irqs) interrupts();
374 uint8_t hb=pWire->read();
375 uint8_t lb=pWire->read();
376 uint16_t data=(hb<<8) | lb;
377 *pData=data;
378 return true;
379 }

511 double calculateLux(uint16_t ch0, uint16_t ch1) {
517 if (ch0==0) return 0;
518 double lux;
519 double ratio = (double)ch1 / (double)ch0;
520 if (ratio <= 0.5) {
521 lux=0.0304*(double)ch0 - 0.062*(double)ch0*pow(ratio, 1.4);
522 } else if (ratio <= 0.61) {
523 lux=0.0224*(double)ch0 - 0.031*(double)ch1;
524 } else if (ratio <= 0.80) {
525 lux=0.0128*(double)ch0 - 0.0153*(double)ch1;
526 } else if (ratio <= 1.30) {
527 lux=0.00146*(double)ch0 - 0.00112*(double)ch1;
528 } else {
529 lux=0;
530 }
531 return lux;
532 }

556 bool readTSLSensor(double *pIlluminanceCh0, double *pIlluminanceCh1,
557 double *pLux, double *pUnitIlluminance) {
558 *pIlluminanceCh0=0.0;
559 *pIlluminanceCh1=0.0;
560 *pLux=0.0;
561 *pUnitIlluminance=0.0;
562 if (!readTSLSensorMeasurement(0xac, pIlluminanceCh0)) return false;
563 if (!readTSLSensorMeasurement(0xae, pIlluminanceCh1)) return false;
564 *pLux=calculateLux(*pIlluminanceCh0, *pIlluminanceCh1);
565 double ui;
566 if (*pLux>1.0) {
567 ui=log(*pLux)*unitIlluminanceSensitivity;
568 } else {
569 ui=0.0;
570 }
571 if (ui<0.0) ui=0.0;
572 if (ui>1.0) ui=1.0;
573 *pUnitIlluminance=ui;
574 return true;
575 }
