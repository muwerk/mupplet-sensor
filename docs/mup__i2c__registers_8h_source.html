71 bool endTransmission(bool releaseBus) {
76 uint8_t retCode = pWire->endTransmission(releaseBus); // true: release bus, send stop
77 switch (retCode) {
78 case 0:
79 lastError = I2CError::OK;
80 return true;
81 case 1:
82 lastError = I2CError::I2C_WRITE_DATA_TOO_LONG;
83 return false;
84 case 2:
85 lastError = I2CError::I2C_WRITE_NACK_ON_ADDRESS;
86 return false;
87 case 3:
88 lastError = I2CError::I2C_WRITE_NACK_ON_DATA;
89 return false;
90 case 4:
91 lastError = I2CError::I2C_WRITE_ERR_OTHER;
92 return false;
93 case 5:
94 lastError = I2CError::I2C_WRITE_TIMEOUT;
95 return false;
96 default:
97 lastError = I2CError::I2C_WRITE_INVALID_CODE;

99 }
