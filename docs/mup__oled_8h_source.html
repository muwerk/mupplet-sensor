419 void sensorUpdates(String topic, String msg, String originator) {
420 ustd::array<String> msgs;
421 char buf[64];
422#ifdef USE_SERIAL_DBG
423 Serial.print("sensorUpdates ");
424 Serial.println(msg);
425#endif
426 for (uint8_t i=0; i<slots; i++) {
427 if (topic==topics[i]) {
428 String valid_numbers="IPFDT";
429 if (valid_numbers.indexOf(formats[i])!= -1) {
430 vals[i]=msg.toFloat();
431 lastUpdates[i]=time(nullptr);
432 if (vals_init[i]==false) {
433 for (uint8_t j=0; j<HIST_CNT; j++) hists[i][j]=vals[i];
434 } else {
435 for (uint8_t j=0; j<HIST_CNT-1; j++) hists[i][j]=hists[i][j+1];
436 hists[i][HIST_CNT-1]=vals[i];
437 }
438 vals_init[i]=true;
439 dirs[i]=vals[i]-hists[i][0];
440 }
441 if (formats[i]=='S') {
442 lastUpdates[i]=time(nullptr);
443 vals_init[i]=true;
444 }
445 }
446 }
447 for (uint8_t i=0; i<slots; i++) {
448 if (vals_init[i]==true) {
449 msgs[i]="?Format";
450 if (formats[i]=='F') {
451 sprintf(buf,"%.1f",vals[i]);
452 msgs[i]=String(buf);
453 }
454 if (formats[i]=='D') {
455 sprintf(buf,"%.2f",vals[i]);
456 msgs[i]=String(buf);
457 }
458 if (formats[i]=='T') {
459 sprintf(buf,"%.3f",vals[i]);
460 msgs[i]=String(buf);
461 }
462 if (formats[i]=='I') {
463 sprintf(buf,"%d",(int)vals[i]);
464 msgs[i]=String(buf);
465 }
466 if (formats[i]=='P') {
467 sprintf(buf,"%d%%",(int)(vals[i]*100));
468 msgs[i]=String(buf);
469 }
470 if (formats[i]=='S') {
471 msgs[i]=msg;
472 }
473 } else {
474 msgs[i]="NaN";
475 }
476 }
477 updateDisplay(msgs);
478 }
