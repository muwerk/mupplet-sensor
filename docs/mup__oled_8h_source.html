413 ustd::array<String> msgs;

416#ifdef USE_SERIAL_DBG
417 Serial.print("sensorUpdates ");

420 for (uint8_t i=0; i<slots; i++) {
421 if (topic==topics[i]) {
422 if (formats[i]=='F') {
423 vals[i]=msg.toFloat();
424 lastUpdates[i]=time(nullptr);
425 if (vals_init[i]==false) {
426 for (uint8_t j=0; j<HIST_CNT; j++) hists[i][j]=vals[i];
427 } else {
428 for (uint8_t j=0; j<HIST_CNT-1; j++) hists[i][j]=hists[i][j+1];
429 hists[i][HIST_CNT-1]=vals[i];
430 }
431 vals_init[i]=true;
432 dirs[i]=vals[i]-hists[i][0];
433 disp_update=true;
434 }
435 if (formats[i]=='S') {
436 lastUpdates[i]=time(nullptr);
437 vals_init[i]=true;
438 disp_update=true;
439 }
440 }
441 }
442 if (disp_update==true) {
443 for (uint8_t i=0; i<slots; i++) {
444 if (vals_init[i]==true) {
445 msgs[i]="?Format";
446 if (formats[i]=='F') {
447 sprintf(buf,"%.1f",vals[i]);
448 msgs[i]=String(buf);
449 }
450 if (formats[i]=='S') {
451 msgs[i]=msg;
452 }
453 } else {
454 msgs[i]="NaN";
455 }
456 }
457 updateDisplay(msgs, dirs);
458 }
