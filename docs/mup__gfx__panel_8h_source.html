696 // This is called on Sensor-update events
697 void sensorUpdates(String topic, String msg, String originator) {
698 ustd::array<String> msgs;
699 char buf[64];
700#ifdef USE_SERIAL_DBG
701 Serial.print("sensorUpdates ");
702 Serial.println(msg);
703#endif
704 for (uint8_t i=0; i<slots; i++) {
705 if (topic==topics[i]) {
706 String valid_numbers="IPFDT";
707 if (valid_numbers.indexOf(formats[i])!= -1) {
708 vals[i]=msg.toFloat();
709 lastUpdates[i]=time(nullptr);
710 if (vals_init[i]==false) {
711 for (uint8_t j=0; j<HIST_CNT; j++) hists[i][j]=vals[i];
712 } else {
713 for (uint8_t j=0; j<HIST_CNT-1; j++) hists[i][j]=hists[i][j+1];
714 hists[i][HIST_CNT-1]=vals[i];
715 }
716 vals_init[i]=true;
717 dirs[i]=vals[i]-hists[i][0];
718 }
719 if (formats[i]=='S') {
720 lastUpdates[i]=time(nullptr);
721 vals_init[i]=true;
722 svals[i]=msg;
723 }
724 }
725 }
726 for (uint8_t i=0; i<slots; i++) {
727 if (vals_init[i]==true) {
728 msgs[i]="?Format";
729 if (formats[i]=='S') {
730 msgs[i]=svals[i];
731 }
732 if (formats[i]=='F') {
733 sprintf(buf,"%.1f",vals[i]);
734 msgs[i]=String(buf);
735 }
736 if (formats[i]=='D') {
737 sprintf(buf,"%.2f",vals[i]);
738 msgs[i]=String(buf);
739 }
740 if (formats[i]=='T') {
741 sprintf(buf,"%.3f",vals[i]);
742 msgs[i]=String(buf);
743 }
744 if (formats[i]=='I') {
745 sprintf(buf,"%d",(int)vals[i]);
746 msgs[i]=String(buf);
747 }
748 if (formats[i]=='P') {
749 sprintf(buf,"%d%%",(int)(vals[i]*100));
750 msgs[i]=String(buf);
751 }
752 } else {
753 msgs[i]="NaN";
754 }
755 }
756 updateDisplay(msgs);
757 }
