151 uint16_t rgbColor(uint8_t red, uint8_t green, uint8_t blue) {
152 switch (displayType) {
153 case DisplayType::SSD1306: // Black or white
154 if (RGB(red, green, blue)==bgColor) {
155 if (isLightTheme) return SSD1306_WHITE;
156 else return SSD1306_BLACK;
157 } else {
158 if (isLightTheme) return SSD1306_BLACK;
159 else return SSD1306_WHITE;
160 }
161 case DisplayType::ST7735: // RGB565 standard
162 return (((red & 0xf8)<<8) + ((green & 0xfc)<<3)+(blue>>3));
163 default:
164 return 0;
165 }
166 }

603 uint32_t relRGB(uint8_t r,uint8_t g, uint8_t b, float brightness, float contrast) {
604 int16_t rt, gt, bt;
605 rt=(int16_t)((((float)r-128.0f)*contrast*2.0f+128.0f)*brightness*2.0f);
606 if (rt>(int16_t)0xff) rt=0xff;
607 if (rt<(int16_t)0) rt=0;
608 gt=(int16_t)((((float)g-128.0f)*contrast*2.0f+128.0f)*brightness*2.0f);
609 if (gt>(int16_t)0xff) gt=0xff;
610 if (gt<(int16_t)0) gt=0;
611 bt=(int16_t)((((float)b-128.0f)*contrast*2.0f+128.0f)*brightness*2.0f);
612 if (bt>(int16_t)0xff) bt=0xff;
613 if (bt<(int16_t)0) bt=0;
614 return GfxDrivers::RGB((uint8_t)rt, (uint8_t)gt, (uint8_t)bt);
615 }

685 bool splitCombinedLayout(String combined_layout) {
686 bool layout_valid=true;
687 bool parsing=true;
688
689 formats="";
690 layout="";
691 slots=0;
692 String line="";
693
694 while (parsing) {
695 int ind = combined_layout.indexOf('|');
696 if (ind==-1) {
697 line=combined_layout;
698 combined_layout="";
699 } else {
700 line=combined_layout.substring(0,ind);
701 combined_layout=combined_layout.substring(ind+1);
702 }
703 for (char c : line) {
704 if (valid_formats_small.indexOf(c)==-1 && valid_formats_long.indexOf(c)==-1) {
705 layout_valid=false;
706 parsing=false;
707 break;
708 } else {
709 ind=valid_formats_small.indexOf(c);
710 if (ind!=-1) {
711 c=valid_formats_long[ind];
712 layout+="S";
713 } else {
714 layout+="L";
715 }
716 formats+=c;
717 ++slots;
718 }
719 }
720 if (!layout_valid) break;
721 if (combined_layout!="") {
722 layout+="|";
723 } else {
724 parsing=false;
725 }
726 }
727
728 lastRefresh=0;
729 delayedUpdate=false;
730 return layout_valid;
731 }

775 bool config2slot(uint16_t slot) {
780 if (slot>=slots) {
781 return false;
782 }
783 pSlots[slot].slotX=0;
784 pSlots[slot].slotY=0;
785 uint16_t ind=0;
786 for (auto c : layout) {
787 if (c=='S') {
788 if (ind==slot) {
789 pSlots[slot].slotLenX=1;
790 pSlots[slot].slotLenY=1;
791 break;
792 } else {
793 pSlots[slot].slotX++;
794 }
795 ++ind;
796 } else if (c=='L') {
797 if (ind==slot) {
798 pSlots[slot].slotLenX=2;
799 pSlots[slot].slotLenY=1;
800 break;
801 } else {
802 pSlots[slot].slotX+=2;
803 }
804 ++ind;
805 } else if (c=='|') {
806 pSlots[slot].slotY++;
807 pSlots[slot].slotX=0;
808 } else {
809 return false;
810 }
811 }
812 pSlots[slot].histLen=defaultHistLen;
813 pSlots[slot].offset=0.0;
814 pSlots[slot].scalingFactor=1.0;
815 switch (formats[slot]) {
816 case 'I':
817 pSlots[slot].slotType=SlotType::NUMBER;
818 pSlots[slot].digits=0;
819 break;
820 case 'F':
821 pSlots[slot].slotType=SlotType::NUMBER;
822 pSlots[slot].digits=1;
823 break;
824 case 'D':
825 pSlots[slot].slotType=SlotType::NUMBER;
826 pSlots[slot].digits=2;
827 break;
828 case 'T':
829 pSlots[slot].slotType=SlotType::NUMBER;
830 pSlots[slot].digits=3;
831 break;
832 case 'S':
833 pSlots[slot].slotType=SlotType::TEXT;
834 pSlots[slot].digits=3;
835 pSlots[slot].histLen=0;
836 break;
837 case 'P':
838 pSlots[slot].slotType=SlotType::NUMBER;
839 pSlots[slot].scalingFactor=100.0;
840 pSlots[slot].digits=1;
841 break;
842 case 'G':
843 pSlots[slot].slotType=SlotType::GRAPH;
844 pSlots[slot].digits=3;
845 break;
846 default:
847 return false;
848 }
849 if (pSlots[slot].histLen) {
850 pSlots[slot].pHist=(float *)malloc(sizeof(float)*pSlots[slot].histLen);
851 pSlots[slot].histInit=false;
852 if (pSlots[slot].pHist) {
853 for (uint16_t j=0; j<pSlots[slot].histLen; j++) {
854 pSlots[slot].pHist[j]=0;
855 }
856 } else {
857 pSlots[slot].histLen=0;
858 return false;
859 }
860 } else {
861 pSlots[slot].pHist=nullptr;
862 pSlots[slot].histLen=0;
863 }
864 pSlots[slot].topic=topics[slot];
865 pSlots[slot].caption=captions[slot];
866 pSlots[slot].lastUpdate=time(nullptr);
867 pSlots[slot].lastHistUpdate=0;

1282 bool displaySlot(uint16_t slot) {
1283 if (slot>=slots) return false;
1284
1285 uint8_t x0=0, y0=0, x1=0, y1=0, xa=0, ya=0;
1286 uint8_t xm0, ym0, xm1,ym1;
1287
1288 // Blank
1289 uint16_t xf0, yf0, xl, yl;
1290 xf0=pSlots[slot].slotX*slotResX;
1291 xl=slotResX*pSlots[slot].slotLenX;
1292 yf0=pSlots[slot].slotY*slotResY+1;
1293 yl=slotResY*pSlots[slot].slotLenY-1;
1294 pDisplay->fillRect(xf0, yf0, xl, yl, pSlots[slot].bgColor);
1295 // Caption font start x0,y0
1296 x0=pSlots[slot].slotX*slotResX+14;
1297 y0=pSlots[slot].slotY*slotResY+3;
1298 x1=pSlots[slot].slotX*slotResX+14;
1299 y1=pSlots[slot].slotY*slotResY+slotResY-3;
1300 xa=pSlots[slot].slotX*slotResX+5;
1301 ya=pSlots[slot].slotY*slotResY+14;
1302 xm0=pSlots[slot].slotX*slotResX+1;
1303 ym0=pSlots[slot].slotY*slotResY+1;
1304 xm1=(pSlots[slot].slotX+1)*slotResX-2+(pSlots[slot].slotLenX-1)*slotResX;
1305 ym1=(pSlots[slot].slotY+1)*slotResY-2+(pSlots[slot].slotLenY-1)*slotResY;
1306
1307 // caption
1308 pDisplay->setFont();
1309 pDisplay->setTextColor(defaultAccentColor);
1310 pDisplay->setTextSize(1);
1311 String first="", second="";
1312 boldParser(pSlots[slot].caption, first, second);
1313 pDisplay->setCursor(x0,y0);
1314 pDisplay->println(first);
1315 pDisplay->setCursor(x0+1,y0);
1316 pDisplay->println(second);
1317
1318 float gmin, gmax, dmin, dmax;
1319 float avg, navg;
1320 dmin= 100000.0;
1321 dmax=-100000.0;
1322 gmax=-100000.0;
1323 gmin= 100000.0;
1324 avg=0.0;
1325 navg=0.0;
1326 if (pSlots[slot].slotType!=SlotType::TEXT) {
1327 if (pSlots[slot].pHist && pSlots[slot].histLen) {
1328 for (uint16_t x=0; x<pSlots[slot].histLen; x++) {
1329 if (pSlots[slot].pHist[x]>gmax) gmax=pSlots[slot].pHist[x];
1330 if (pSlots[slot].pHist[x]<gmin) gmin=pSlots[slot].pHist[x];
1331 uint16_t backView=0;
1332 if (pSlots[slot].histLen>=10) backView=pSlots[slot].histLen-10;
1333 if (x>=backView) {
1334 avg += pSlots[slot].pHist[x];
1335 navg++;
1336 if (pSlots[slot].pHist[x]>dmax) dmax=pSlots[slot].pHist[x];
1337 if (pSlots[slot].pHist[x]<dmin) dmin=pSlots[slot].pHist[x];
1338 }
1339 }
1340 float ref;
1341 if (navg>0.0) ref=avg/navg;
1342 else ref=0.0;
1343 pSlots[slot].deltaDir=pSlots[slot].currentValue-ref;
1344 }
1345 }
1346 if (pSlots[slot].slotType!=SlotType::GRAPH) {
1347 // Main text
1348 pDisplay->setFont(&FreeSans12pt7b);
1349 pDisplay->setTextColor(defaultColor);
1350 pDisplay->setTextSize(1);
1351 pDisplay->setCursor(x1,y1);
1352 pDisplay->println(pSlots[slot].currentText);
1353 pDisplay->setCursor(x1+1,y1);
1354 pDisplay->println(pSlots[slot].currentText);
1355 if (pSlots[slot].slotType!=SlotType::TEXT) {
1356 // arrow
1357 if (pSlots[slot].deltaDir != 0.0) {
1358 drawArrow(xa,ya,(pSlots[slot].deltaDir>0.0),8,3,7);
1359 }
1360 }
1361 } else {
1362 // Graph
1363 if (pSlots[slot].pHist && pSlots[slot].histLen) {
1364 double deltaY=gmax-gmin;
1365 if (deltaY<0.0001) deltaY=1;
1366 double deltaX=(double)(xm1-xm0)/(double)(pSlots[slot].histLen);
1367 int lx0,ly0,lx1,ly1;
1368 int gHeight=(ym1-ym0)-11; // font size of caption.
1369 for (uint16_t i=1; i<pSlots[slot].histLen; i++) {
1370 lx0=xm0+(int)((double)(i-1)*deltaX); lx1=xm0+(int)((double)i*deltaX);
1371 ly0=ym1-(int)((pSlots[slot].pHist[i-1]-gmin)/deltaY*(double)(gHeight));
1372 ly1=ym1-(int)((pSlots[slot].pHist[i]-gmin)/deltaY*(double)(gHeight));
1373 uint32_t col;
1374 if (ly1<ly0) col=defaultIncreaseColor;
1375 else {
1376 if (ly1==ly0) col=defaultConstColor;
1377 else col=defaultDecreaseColor;
1378 }
1379 pDisplay->drawLine(lx0, ly0, lx1, ly1, col);
1380 }
1381 }
1382 }
1383 pSlots[slot].hasChanged=false;
1384 return true;
1385 }

1424 bool updateSlot(uint16_t slot, String msg) {
1425 if (slot>=slots) return false;
1426 if (timeDiff(pSlots[slot].lastFrame, millis()) < pSlots[slot].frameRate) return false;
1427 bool changed=false;
1428 float k=pSlots[slot].scalingFactor;
1429 float o=pSlots[slot].offset;
1430 switch (pSlots[slot].slotType) {
1431 case SlotType::TEXT:
1432 if (pSlots[slot].currentText!=msg) {
1433 changed=true;
1434 }
1435 pSlots[slot].currentText=msg;
1436 pSlots[slot].isValid=true;
1437 pSlots[slot].lastUpdate=time(nullptr);
1438 pSlots[slot].lastFrame=millis();
1439 break;
1440 case SlotType::NUMBER:
1441 case SlotType::GRAPH:
1442 pSlots[slot].currentValue=msg.toFloat()*k+o;
1443 pSlots[slot].isValid=true;
1444 pSlots[slot].lastUpdate=time(nullptr);
1445 pSlots[slot].lastFrame=millis();
1446 String newVal=String(pSlots[slot].currentValue,(uint16_t)pSlots[slot].digits);
1447 if (pSlots[slot].currentText!=newVal) {
1448 changed=true;
1449 }
1450 pSlots[slot].currentText=newVal;
1451 if (pSlots[slot].pHist && pSlots[slot].histLen>0) {
1452 if (!pSlots[slot].histInit) {
1453 for (uint16_t i=0; i<pSlots[slot].histLen; i++) {
1454 pSlots[slot].pHist[i]=pSlots[slot].currentValue;
1455 }
1456 pSlots[slot].lastHistUpdate=millis();
1457 pSlots[slot].histInit=true;
1458 changed=true;
1459 } else {
1460 while (timeDiff(pSlots[slot].lastHistUpdate, millis())>pSlots[slot].histSampleRateMs) {
1461 for (uint16_t i=0; i<pSlots[slot].histLen-1; i++) {
1462 pSlots[slot].pHist[i]=pSlots[slot].pHist[i+1];
1463 }
1464 pSlots[slot].lastHistUpdate += pSlots[slot].histSampleRateMs;
1465 pSlots[slot].pHist[pSlots[slot].histLen-1]=pSlots[slot].currentValue;
1466 changed=true;
1467 }
1468 pSlots[slot].pHist[pSlots[slot].histLen-1]=pSlots[slot].currentValue;
1469 }
1470 }
1471 break;
1472 }
1473 pSlots[slot].hasChanged=changed;
1474 return changed;
1475 }
